def privacy
    #get_test_by_id
    #check_test_owner
    if @test.change_privacy
      flash[:success] ="This test is #{@test.private? ? 'Private' : 'Public'} now"
      redirect_to test_path(@test)
    else
      flash[:danger] = 'Unable to change privacy,Try later'
      redirect_to test_path(@test)
    end
end
